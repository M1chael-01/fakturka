***** FAKTURKA CASHFLOW *******
CREATE DATABASE fakturka_cashflow
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LOCALE_PROVIDER = 'libc'
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;


CREATE TABLE public.one_cashflow
(
    id numeric,
    "userID" numeric,
    operation text,
    date date,
    description text,
    payment text,
    amount numeric,
    categorie text,
    note text,
    PRIMARY KEY (id)
);

ALTER TABLE IF EXISTS public.one_cashflow
    OWNER to postgres;
